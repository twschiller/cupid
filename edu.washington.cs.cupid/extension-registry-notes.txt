private IExtensionRegistry registry = Platform.getExtensionRegistry();

for (String namespace : registry.getNamespaces()){
	System.out.println("Namespace: " + namespace);
	for (IExtension extension : registry.getExtensions(namespace)){
		if (extension.isValid()){
			System.out.println("id: " + id(extension) + " extends " + extension.getExtensionPointUniqueIdentifier());
			IContributor contributor = extension.getContributor();
		}
	}
}
		
public static String id(IExtension extension){
	if (extension.getUniqueIdentifier() != null){
		return extension.getUniqueIdentifier();
	}else if (extension.getSimpleIdentifier() != null){
		return extension.getSimpleIdentifier();
	}else if (extension.getLabel() != null){
		return extension.getLabel();
	}else{
		return extension.toString();
	}
}
